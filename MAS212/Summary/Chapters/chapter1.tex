\dfn{Field}{
    Algebraic structure $\bbF$ satisfying given properties are called field:
    \begin{enumerate}
        \item Addition is commutative: $\forall\{x,y\}\subset\bbF\s(x+y=y+z)$
        \item Addition is associative: $\forall\{x,y,z\}\subset\bbF\s(x+(y+z)=(x+y)+z)$
        \item $\forall x\in\bbF\s\ues0\in\bbF\s(x+0=x)$ 
        \item $\forall x\in\bbF\s\ues(-x)\in\bbF\s(x+(-x)=0)$
        \item Multiplication is commutative: $\forall\{x,y\}\subset\bbF\s(xy=yx)$
        \item Multiplication is associative: $\forall\{x,y,z\}\subset\bbF\s(x(yz)=(xy)z)$
        \item $\forall x\in\bbF\s\ues1\in\bbF\s(x1=x)$
        \item $\forall x\in\bbF\s\ues x\inv=1/x\in\bbF\s(xx\inv=1)$
        \item $\forall\{x,y,z\}\in\bbF\s(x(y+z)=xy+xz)$
    \end{enumerate}
}
